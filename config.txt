#
# 1. Crie a aba:
#    [aba:downloads2]
#    tipo_fixo = downloads
#    icone = 🎮
#    nome = Jogos
#    nome_en = Games

# 2. Crie as páginas com  aba = downloads2 :
#    [pagina:minecraft]
#    aba = minecraft
#    nome = Minecraft
#    ...
#
# 3. Crie os itens normalmente com  pagina = minecraft
